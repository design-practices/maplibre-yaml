Let's create a map showing earthquake data in under 5 minutes using the `<ml-map>` web component.

## 1. Create an HTML file

Create a new file called `index.html`:

```html
<!DOCTYPE html>
<html>
  <head>
    <title>My First Map</title>
    <link
      href="https://unpkg.com/maplibre-gl/dist/maplibre-gl.css"
      rel="stylesheet"
    />
    <script type="module">
      import "https://unpkg.com/@maplibre-yaml/core/register";
    </script>
    <style>
      body {
        margin: 0;
      }
      ml-map {
        display: block;
        width: 100vw;
        height: 100vh;
      }
    </style>
  </head>
  <body>
    <ml-map src="earthquake-map.yaml"></ml-map>
  </body>
</html>
```

## 2. Create your YAML configuration

Create a file called `earthquake-map.yaml`:

```yaml
type: map
id: earthquake-map
config:
  center: [-120, 37]
  zoom: 5
  mapStyle: "https://demotiles.maplibre.org/style.json"
layers:
  - id: earthquakes
    type: circle
    source:
      type: geojson
      url: "https://earthquake.usgs.gov/earthquakes/feed/v1.0/summary/all_week.geojson"
    paint:
      circle-radius: 6
      circle-color: "#e74c3c"
      circle-stroke-width: 1
      circle-stroke-color: "#ffffff"
```

## 3. Run a local server

```bash
npx serve .
```

Open http://localhost:3000 in your browser.

## Add interactivity

Let's make the earthquakes clickable. Update your YAML:

```yaml {18-24}
type: map
id: earthquake-map
config:
  center: [-120, 37]
  zoom: 5
  mapStyle: "https://demotiles.maplibre.org/style.json"
layers:
  - id: earthquakes
    type: circle
    source:
      type: geojson
      url: "https://earthquake.usgs.gov/earthquakes/feed/v1.0/summary/all_week.geojson"
    paint:
      circle-radius: 6
      circle-color: "#e74c3c"
      circle-stroke-width: 1
      circle-stroke-color: "#ffffff"
    interactive:
      hover:
        cursor: pointer
      click:
        popup:
          - h3:
              - property: title
```

## Add live updates

Make the map refresh automatically every 60 seconds:

```yaml {6-9}
layers:
  - id: earthquakes
    type: circle
    source:
      type: geojson
      url: "https://earthquake.usgs.gov/earthquakes/feed/v1.0/summary/all_hour.geojson"
      refresh:
        refreshInterval: 60000
        updateStrategy: replace
    paint:
      circle-radius: 6
      circle-color: "#e74c3c"
```
